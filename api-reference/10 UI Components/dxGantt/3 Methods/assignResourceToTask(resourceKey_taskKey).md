---
##### shortDescription
Assigns a resource to a task.

##### param(resourceKey): Object
The resource key.

##### param(taskKey): Object
The task key.

---

---

##### jQuery

    <!-- tab: index.js -->
    var gantt = $("#ganttContainer").dxGantt("instance");
    gantt.assignResourceToTask("resource_key","task_key");
