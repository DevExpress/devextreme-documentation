---
id: dxMap.Options.key
type: String | Object
default: ''
---
---
##### shortDescription
A key used to authenticate the application within the required map provider.

---
---
##### jQuery

    <!-- tab: index.js -->
    $(function() {
        $("#mapContainer").dxMap({
            // ...
            key: {
                bing: "MY_BING_MAPS_KEY",
                google: "MY_GOOGLE_MAPS_KEY",
                googleStatic: "MY_GOOGLE_STATIC_MAPS_KEY"
            }
        });
    });
