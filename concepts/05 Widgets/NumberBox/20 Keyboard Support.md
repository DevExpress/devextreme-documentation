    <tr>

        <th>Action</th>

    </tr>
    <tr>
        <td>&darr; or mouse wheel down</td>
        <td>Decreases the widget value by a <a href="/Documentation/ApiReference/UI_Widgets/dxNumberBox/Configuration/#step">step</a>.</td>
    </tr>
</table>
